scripted_gui = {
    KORNA_open_button = {
        context_type = player_context
        parent_window_token = politics_tab
        window_name = "countrypoliticsview_KORNA_open_button_window"
		dirty = KORNAscreen_open

        visible = {
           tag = KOR
		   has_country_flag = KOR_NA_ACTIVE
        }

        effects = {
            politicsview_KORNA_button_click = {
                if = {
					limit = { has_variable = KORNAscreen_open }
					clear_variable = KORNAscreen_open
				}
				else = {
					set_variable = { KORNAscreen_open = 1 }
				}
            }
        }
    }
	countrypoliticsview_KORNA_main = {
        context_type = player_context
        parent_window_token = politics_tab
        window_name = "countrypoliticsview_KORNA_main"
		dirty = KORNAscreen_open

        visible = {
			has_country_flag = has_symbolic_figurehead
			check_variable = { KORNAscreen_open > 0 }
        }
    }
    politicsview_KORNA_button = {
        context_type = player_context
        parent_window_token = politics_tab
        window_name = "politicsview_KORNA_button"
		dirty = KORNAscreen_open

        visible = {
			tag = KOR
			has_country_flag = KOR_NA_ACTIVE
			check_variable = { KORNAscreen_open > 0 }
        }
    }
}
