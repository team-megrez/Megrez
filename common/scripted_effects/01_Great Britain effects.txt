### Great Britain / England ###

create_GBP_cores = {
	ENG = {
		every_core_state = {
			limit = {
				NOT = {
					AND = {
						is_core_of = SCO
						has_variable = GBP_no_scotland
					}
					AND = {
						is_core_of = WLS
						has_variable = GBP_no_wales
					}
				}
			}
			add_core_of = GBP
		}
	}
}
