LEB_set_up_chehab_coup = {
	if = {
		limit = {
			LEB_fuad_chehab = {
				NOT = {
					is_character_slot = political_advisor
				}
			}
		}
		add_advisor_role = {
			character = LEB_fuad_chehab
			activate = yes
			advisor = {
				slot = political_advisor
				idea_token = LEB_fuad_chehab_pa
				traits = { KR_respected_war_hero }
				cost = 0
				can_be_fired = no
			}
		}
	}
	activate_mission = LEB_fuad_chehab_intervenes
}
