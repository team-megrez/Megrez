		limit = {
			check_variable = { pp_spent > pp_need }
			check_variable = { current_level < 10 }
		}
		set_temp_variable = { ethnic_situation = 1 } # 1 = +1 relations level
	}
	else_if = {
		limit = {
			check_variable = { pp_spent < pp_need }
			check_variable = { current_level > 1 }
		}
		set_temp_variable = { ethnic_situation = -1 } # -1 = -1 relations level
	}
	else = {
		set_temp_variable = { ethnic_situation = 0 }
	}
}
