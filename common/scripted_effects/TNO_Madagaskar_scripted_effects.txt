MAD_setup_civil_war = {
	hidden_effect = {
		MAD = {
			leave_faction = yes
			end_puppet = GER
			every_owned_state = {
				set_demilitarized_zone = no
				add_core_of = MDG
				add_core_of = MDM
			}
			add_manpower = 300000
			MAD_Erhard_Milch = { set_nationality = MDG }
			MAD_Werner_Goldberg = { set_nationality = MDG }
			MAD_Walter_Hollaender = { set_nationality = MDG }
			MAD_Jurgen_Wattenberg = { set_nationality = MDG }
			MAD_Gabriel_Ramanantsoa = { set_nationality = MDM }

			add_ideas = { MAD_Horst_von_Oppenfeld_sec }
			delete_unit_template_and_units = {
				division_template = "SS-Division"
			}
			delete_unit_template_and_units = {
				division_template = "Garnision Madagaskar"
			}
			#set_country_flag = MAD_civil_war

			remove_ideas = MAD_idea_mismanaged_rebel_activity

			load_oob = MAD_CIVIL_WAR_5
			if = {
				limit = {
					has_country_flag = MAD_OFN_support
				}
				USA = {
					# Increase OFN volunteers to MAD
				}
			}
		}

		

		MDM = { # Malagasy Rebels
			add_manpower = 300000
			transfer_state = 543

			promote_character = MAD_Gabriel_Ramanantsoa
			add_ideas = {
				MDM_Gabriel_Ramanantsoa_hog
				MDM_Jacques_Rabemananjara_for
				MDM_Philibert_Tsiranana_eco
				MDM_Didier_Ratsiraka_sec
			}

			load_oob = MDM_CIVIL_WAR_5
			
		}

		MDG = { # German Garrison
			add_manpower = 300000
			transfer_state = 1862
			transfer_state = 1043
			transfer_state = 1877

MAD_launch_civil_war = {
	hidden_effect = {
		every_country = {
			news_event = {
				id = WORLD.67
				days = 1
			}
		}
		USA = {
			TNO_fopo_add_madagascar_conflict_tab = yes
			set_temp_variable = { target = MAD }
			TNO_increase_volunteer_cap_1 = yes
		}
		JAP = {
			TNO_fopo_add_madagascar_conflict_tab = yes
			set_temp_variable = { target = MDM }
			TNO_increase_volunteer_cap_1 = yes
		}
		GER = {
			TNO_fopo_add_madagascar_conflict_tab = yes
			set_temp_variable = { target = MDG }
			TNO_increase_volunteer_cap_1 = yes
		}

		#MAD = {
		#	add_named_threat = {
		#		threat = 5
		#		name = threat.22.t
		#	}
		#}

		MDM = {
			declare_war_on = {
				target = MAD
				type = civil_war
			}
		}

		MDG = {
			declare_war_on = {
				target = MAD
				type = civil_war
			}
			declare_war_on = {
				target = MDM
				type = civil_war
			}
		}
	}
}

MAD_check_civil_war_status = {
	if = {
		limit = {
			country_exists = MAD
			NOT = { country_exists = MDM }
			NOT = { country_exists = MDG }
		}
		USA = { country_event = MADAGASKAR.9 }
		MAD = {
			every_owned_state = {
				remove_core_of = MDM
				remove_core_of = MDG
			}
		}
	}
	else_if = {
		limit = {
			country_exists = MDM
			NOT = { country_exists = MAD }
			NOT = { country_exists = MDG }
		}
		JAP = { country_event = MADAGASKAR.10 }
		MDM = {
			every_owned_state = {
				remove_core_of = MAD
				remove_core_of = MDG
			}
		}
	}
	else_if = {
		limit = {
			country_exists = MDG
			NOT = { country_exists = MAD }
			NOT = { country_exists = MDM }
		}
		GER = { country_event = MADAGASKAR.11 }
		MDG = {
			every_owned_state = {
				remove_core_of = MAD
				remove_core_of = MDM
			}
		}
	}
	else = {
		log = "[GetDateText]: Madagascar war keep going"
	}
}
