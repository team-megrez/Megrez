					has_country_flag = {

					}
				}
				country_event = { id = uob.7 days = 20 }
			}
			else_if = {
				limit = {
					has_country_flag = {
						flag = ENG_tuc_federalist
						value > 1
					}
				}
				country_event = { id = uob.8 days = 20 }
			}
			else_if = {
				limit = {
					has_country_flag = {
						flag = ENG_tuc_autonomist
						value > 1
					}
				}
				country_event = { id = uob.9 days = 20 }
			}
			else = {
				country_event = { id = uob.10 days = 20 }
			}
		}
	}
}
