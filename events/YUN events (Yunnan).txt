country_event = {# Long Yun Takes Over!
	id = yun.2
	title = yun.2.t
	desc = yun.2.d
	picture = GFX_report_event_YUN_long_yun_portrait

	is_triggered_only = yes
	fire_only_once = yes

	immediate = {
		hidden_effect = {
			YUN_long_yun = { set_nationality = ROOT }
			YUN_hu_ruoyu = { set_nationality = ROOT }
			YUN_lu_han = { set_nationality = ROOT }
		}
	}

	option = {# Long Live Long Yun!
		log = "[GetLogInfo]: event yun.2 option yun.2.a"
		name = yun.2.a
		add_stability = -0.10
		add_ideas = YUN_chaos_army
		retire_character = YUN_tang_jiyu
		activate_advisor = YUN_zhang_chong_sic
		hidden_effect = {
			clr_country_flag = YUN_TT_time_has_run_out
			if = {
				limit = {
					has_idea = YUN_Tangs_rule
				}
				remove_ideas = YUN_Tangs_rule
			}
			if = {
				limit = {
					has_idea = YUN_Tangs_rule2
				}
				remove_ideas = YUN_Tangs_rule2
			}
			if = {
				limit = {
					has_idea = YUN_Tangs_rule3
				}
				remove_ideas = YUN_Tangs_rule3
			}
			if = {
				limit = {
					has_idea = YUN_Tangs_rule_about_to_die
				}
				remove_ideas = YUN_Tangs_rule_about_to_die
			}
			if = {
				limit = {
					has_idea = YUN_this_is_the_end_for_you_Tang
				}
				remove_ideas = YUN_this_is_the_end_for_you_Tang
			}
			if = {
				limit = {
					has_idea = YUN_this_is_the_end_for_you_Tang2
				}
				remove_ideas = YUN_this_is_the_end_for_you_Tang2
			}
			if = {
				limit = {
					has_idea = YUN_Tangs_rule_isolation
				}
				remove_ideas = YUN_Tangs_rule_isolation
			}
			if = {
				limit = {
					has_idea = YUN_Tangs_rule_isolation2
				}
				remove_ideas = YUN_Tangs_rule_isolation2
			}

			load_focus_tree = { tree = YUN_Swapped_tree }
			if = {
				limit = { has_war = yes } #if at war, Zhu De intervenes immediately
				country_event = { id = yun.29 days = 3 }
			}
			else = {
				random_list = {
					1 = { country_event = { id = yun.3 days = 7 } } #First Chain
					1 = { country_event = { id = yun.5 days = 7 } } #Second Chain
					1 = { country_event = { id = yun.7 days = 7 } } #Third Chain
				}
			}
			YUN_long_yun = {
				if = {
					limit = { has_ideology = paternal_autocrat_subtype }
					promote_character = paternal_autocrat_subtype
				}
				else = {
					add_country_leader_role = {
						country_leader = {
							desc = YUN_long_yun_leader_desc
							ideology = paternal_autocrat_subtype
						}
						promote_leader = yes
					}
				}
			}
			YUN_hu_ruoyu = {
				if = {
					limit = { has_ideology = paternal_autocrat_subtype }
					promote_character = paternal_autocrat_subtype
				}
				else = {
					add_country_leader_role = {
						country_leader = {
							desc = YUN_hu_ruoyu_leader_desc
							ideology = paternal_autocrat_subtype
						}
					}
				}
			}
			YUN_lu_han = {
				if = {
					limit = { has_ideology = paternal_autocrat_subtype }
					promote_character = paternal_autocrat_subtype
				}
				else = {
					add_country_leader_role = {
						country_leader = {
							desc = YUN_lu_han_leader_desc
							ideology = paternal_autocrat_subtype
						}
					}
				}
			}
			every_character = {
				limit = {
					OR = {
						is_character = YUN_tang_jiyao
						is_character = YUN_tang_jiyu
						has_character_flag = SZC_officer_leader
					}
				}
				retire = yes
			}
			if = {
				limit = {
					has_war = no
				}
				603 = { set_demilitarized_zone = yes }
				816 = { set_demilitarized_zone = yes }
				1051 = { set_demilitarized_zone = yes }
			}
			delete_unit = { disband = no }
			set_party_name = {
				ideology = paternal_autocrat
				long_name = YUN_Long_Yun_takes_over_party_long
				name = YUN_Long_Yun_takes_over_party
			}
			set_party_name = {
				ideology = social_liberal
				long_name = YUN_Zhu_Peide_takes_over_party_long
				name = YUN_Zhu_Peide_takes_over_party
			}
			set_party_name = {
				ideology = social_conservative
				long_name = YUN_Zhu_Peide_takes_over_party_long
				name = YUN_Zhu_Peide_takes_over_party
			}
			set_party_name = {
				ideology = authoritarian_democrat
				long_name = YUN_Zhu_Peide_takes_over_party_long
				name = YUN_Zhu_Peide_takes_over_party
			}
		}
		remove_puppet_and_leave_faction = yes
		news_event = { id = yun.600 days = 1 }
	}
}
