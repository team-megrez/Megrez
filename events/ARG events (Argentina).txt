country_event = { # loop event for unit desertions during ARG-CHL war (donotlog)
	id = arg_chl_war.100
	hidden = yes
	is_triggered_only = yes

	trigger = {
		ARG = { has_war_with = FOP }
	}

	immediate = {
		country_event = { id = arg_chl_war.100 days = 24 random_days = 10 random_hours = 24 }
	}

	option = {
		if = {
			limit = {
				ARG = {
					is_ai = yes
					surrender_progress > 0.4
				}
			}
			ARG = { country_event = arg_chl_war.101 }
		}
		else_if = {
			limit = {
				FOP = {
					is_ai = yes
					surrender_progress > 0.4
				}
			}
			FOP = { country_event = arg_chl_war.102 }
		}
		### Since it doesn't make sense for CHL to join ARG, instead reduce available manpower and inf equipment
		if = {
			limit = {
				CHL = {
					is_ai = yes
					has_war_with = ARG
					surrender_progress > 0.4
				}
			}
			CHL = { country_event = arg_chl_war.103 }

			### Die faster pls, thank you - solves CHL sitting in northern mountains with 100% war support
			if = {
				limit = {
					CHL = { surrender_progress > 0.7 }
				}
				CHL = {
					add_stability = -0.25
					add_war_support = -0.25
					disband_twenty_percent_units_delete_equipment = yes
				}
			}
		}
	}
}

					limit = {
						country_exists = URG
						NOT = {
							has_war_with = URG
							has_subject = URG
						}
					}
					country_event = { id = argcar.30 days = 10 }
				}
				else = {
					country_event = { id = argcar.31 days = 10 }
				}

					limit = {
						PRU = {
							NOT = {
								is_subject_of = ARG

						}
					}

			}
		}

		if = {
			limit = {
				NOT = {
					has_authoritarian_government = yes
				}
			}
			load_focus_tree = KR_Argentina_Democratic
			complete_national_focus = ARG_Rigged_Elections
			set_country_flag = ARG_GOU_in_control
			set_politics = {
				ruling_party = paternal_autocrat
				elections_allowed = yes
				last_election = "1935.9.13"
				election_frequency = 72
			}
			hidden_effect = {
				country_event = { id = argcar.23 days = 3 }
			}

		change_ruling_party_popularity = yes
	}
