country_event = {	#Iraqi Revolt
	id = irqrev.1
	immediate = {
		log = "KR_Event_Logging;Iraq REVOLTS AGAINST OTT;[GetDateText]"
	}
	title = irqrev.1.t
	desc = irqrev.1.d

	picture = GFX_IRQ_Tank_Street

	trigger = {
		has_war_with = EGY
	}

	fire_only_once = yes

	is_triggered_only = yes

	immediate = {
		if = {
			limit = { 291 = { is_owned_and_controlled_by = OTT } }
			IRQ = {
				transfer_technology_without_doctrines = yes
				transfer_state = 291
				add_country_leader_role = {
					character = IRQ_kamil_shabib
					country_leader = {
						ideology = authoritarian_democrat_subtype
					}
					promote_leader = yes
				}
				activate_advisor = IRQ_salah_al_din_al_sabbagh_sic
				set_politics = {
					ruling_party = authoritarian_democrat
					last_election = "1935.10.11"
					election_frequency = 48
					elections_allowed = yes
				}
				set_popularities = {
					totalist = 0
					syndicalist = 0
					radical_socialist = 16
					social_democrat = 20
					social_liberal = 5
					market_liberal = 6
					social_conservative = 3
					authoritarian_democrat = 34
					paternal_autocrat = 15
					national_populist = 1
				}
				declare_war_on = {
					target = OTT
					type = annex_everything
				}
				add_ideas = ott_syrian_uprising
				IRQ_spawn_initial_units = yes
			}
			EGY = {
				add_to_faction = IRQ
			}
		}
	}

	option = {
		log = "[GetLogInfo]: event irqrev.1 option irqrev.1.a"
		name = irqrev.1.a
		every_owned_state = {
			limit = {
				is_core_of = IRQ
			}
			ott_increase_revolt_risk = yes
		}
	}
}
