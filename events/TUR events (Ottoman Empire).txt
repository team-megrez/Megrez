country_event = { #Delete Irregular Divisions
	id = ottgui.1

	hidden = yes

	is_triggered_only = yes

	option = {
		log = "[GetLogInfo]: event ottgui.1 option 1"
		clr_country_flag = created_irregular_template
		delete_unit_template_and_units = { division_template = "Ottoman Irregulars" }
		every_state = {
			limit = { has_state_flag = recruited_irregulars }
			clr_state_flag = recruited_irregulars
		}
	}
}

country_event = { #cheat event for the AI (democratic path)
	id = ottgui.3

	hidden = yes

	is_triggered_only = yes

	option = {
		log = "[GetLogInfo]: event ottgui.3 option 1"
		### Add Turkish Nationalist votes in favour
		custom_effect_tooltip = 30_nat_tur_sup_tt
		add_to_variable = { nat_tur_sup_deputies = 30 }

		### Add Arab Nationalist votes in favour
		custom_effect_tooltip = 30_nat_ara_sup_tt
		add_to_variable = { nat_ara_sup_deputies = 30 }

		### Add Conservative votes in favour
		if = {
			limit = {
				NOT = { has_game_rule = { rule = OTT_path option = OTT_PATH_SOCCON } }
			}
			custom_effect_tooltip = 30_con_sup_tt
			add_to_variable = { con_sup_deputies = 30 }
		}

		### Calculate coalition strength
		OTT_calc_coalition_strength = yes

		### cheat for marlib path
		if = {
			limit = {
				has_game_rule = { rule = OTT_path option = OTT_PATH_MARLIB }
				NOT = { has_government = market_liberal }
			}
			add_popularity = {
				ideology = market_liberal
				popularity = 0.10
			}
		}
	}
}
