	id = romania.46
	title = romania.46.t
	desc = romania.46.d
	picture = GFX_report_event_ger_civil_war

	trigger = {
		OR = {
			has_war_with = 84.owner
			can_declare_war_on = 84.owner
		}
	}

	fire_only_once = yes
	is_triggered_only = yes

	immediate = {
		var:84.owner = {
			save_event_target_as = ROM_transylvania_target
			if = {
				limit = { is_subject = yes }
				overlord = { save_event_target_as = ROM_transylvania_ally }
			}
			else_if = {
				limit = { is_in_faction = yes }
				faction_leader = { save_event_target_as = ROM_transylvania_ally }
			}
		}
	}

	option = {
		log = "[GetLogInfo]: event romania.46 option romania.46.a"
		name = romania.46.a
		set_country_flag = ROM_TRS_revolt
		if = {
			limit = { NOT = { has_war_with = event_target:ROM_transylvania_target } }
			declare_war_on = {
				target = event_target:ROM_transylvania_target
				type = take_state_focus
				generator = { 789 84 76 82 83 }
			}
		}
		effect_tooltip = {
			if = {
				limit = { has_war_with = TRS }
				annex_country = { target = TRS transfer_troops = yes }
			}
			TRS = {
				every_core_state = {
					limit = { owner = { is_ally_with = event_target:ROM_transylvania_target } }
					set_state_owner_to = ROOT
					add_dynamic_modifier = { modifier = ROM_transylvanian_collaboration }
				}
			}
		}
		hidden_effect = {
			every_other_country = {
				limit = { is_ally_with = event_target:ROM_transylvania_target }
				country_event = romania.50
			}
			if = {
				limit = { event_target:ROM_transylvania_target = { original_tag = TRS } }
				annex_country = { target = event_target:ROM_transylvania_target transfer_troops = yes }
			}
			TRS = {
				every_core_state = {
					limit = {
						owner = {
							OR = {
								tag = ROOT
								has_war_with = ROOT
								is_ally_with = event_target:ROM_transylvania_target
								is_ally_with = event_target:ROM_transylvania_ally
							}
						}
						NOT = { state = 875 }
					}

					#Remove AUS/HUN armies
					teleport_armies = {
						limit = {
							OR = {
								has_war_with = ROOT
								is_ally_with = event_target:ROM_transylvania_target
								is_ally_with = event_target:ROM_transylvania_ally
							}
						}
						to_state_array = owned_controlled_states
					}

					#Transfer control
					set_state_owner_to = ROOT
					set_state_province_controller = {
						controller = ROOT
						limit = { has_war_with = ROOT }
					}

					if = {
						limit = { has_resistance = yes }
						set_resistance = 0
					}
					if = {
						limit = { NOT = { state = 789 } } #Szeklerland
						add_dynamic_modifier = { modifier = ROM_transylvanian_collaboration }
					}
				}
			}

			#Strongest possible revolt
			if = {
				limit = {
					check_variable = { ROM.revolt > 8 }
				}
				load_oob = "ROM_TRS_revolt2"
				event_target:ROM_transylvania_target = {
					add_timed_idea = {
						idea = ROM_TRS_revolt
						days = 30
					}
				}
			}
			else_if = {
				limit = {
					check_variable = { ROM.revolt > 6 }
				}
				load_oob = "ROM_TRS_revolt1"
				event_target:ROM_transylvania_target = {
					add_timed_idea = {
						idea = ROM_TRS_revolt
						days = 30
					}
				}
			}
			else_if = {
				limit = {
					check_variable = { ROM.revolt > 4 }
				}
				load_oob = "ROM_TRS_revolt1"
				event_target:ROM_transylvania_target = {
					add_timed_idea = {
						idea = ROM_TRS_revolt
						days = 14
					}
				}
			}
			else_if = {

				}
				event_target:ROM_transylvania_target = {
					add_timed_idea = {

				ROM_great_game

				ROM_sentinels
				ROM_legionary_worker_corps1

				ROM_cross_brotherhoods_idea
			}
			ROM_carol_ii = { retire = yes }
			set_country_flag = ROM_Michael_I
			set_country_flag = ROM_carol_exile
			add_to_variable = { var = ROM.king_influence value = 50 }
			country_event = { id = romania.172 days = 65 }
			country_event = { id = romania.173 days = 126 }
		}
	}
	option = {
		log = "[GetLogInfo]: event romania.240 option romania.240.a"
		name = romania.240.a
		add_political_power = -150
		add_war_support = -0.5
		if = {
			limit = { has_dictatorship_government = yes }
			set_politics = {
				ruling_party = authoritarian_democrat
				elections_allowed = yes
				election_frequency = 48
			}
			set_temp_variable = { new_ideology = token:authoritarian_democrat }
			add_ideas = ROM_rebuilding_the_democracy
		}
		else_if = {
			limit = { has_socialist_government = no }
			set_politics = {
				ruling_party = market_liberal
				elections_allowed = yes
				election_frequency = 48
			}
			set_temp_variable = { new_ideology = token:market_liberal }
			add_ideas = ROM_rebuilding_the_democracy
		}
		if = {
			limit = { is_subject = yes }
			set_political_party = {
				ideology = var:new_ideology
				popularity = 60
			}
		}
		else = {
			add_popularity = {
				ideology = var:new_ideology
				popularity = 0.15
			}
		}
	}
}
